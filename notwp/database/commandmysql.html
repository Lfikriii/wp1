Masuk ke Mysql

1

Masuk ke directory bin
cd C:/xampp/mysql/bin
2

Mengaktifkan Mysql
mysql -u root -p



CREATE

1

Membuat Database
create database {nama_database};
contoh: create database akademik;
2

Menggunakan database
use {nama_database}
contoh: use akademik;
3

Membuat tabel
create table {nama_table} ({colom1 typedata(jumlah)}, {colom2 typedata(jumlah)}, dst..);
contoh: create table mahasiswa (NIM varchar(12), nama varchar(30));
4

Membuat tabel lengkap
create table {nama_table} ({colom1 typedata(jumlah) primary key not null}, {colom2 typedata({jumlah}) not null}, dst..);
contoh: create table mahasiswa (NIM varchar(12) primary key not null, nama varchar(30) not null);



MENAMPILKAN

1

Menampilkan list database
show databases;
2

Menampilkan list tabel
show tables;
3

Menampilkan deskripsi tabel
desc {nama_tabel}
Contoh: desc mahasiswa;
4

Menampilkan isi tabel
select * from {nama_tabel};
Contoh: select * from mahasiswa;



ALTER

1

Menambahkan primary key
alter table {nama_tabel} add primary key ({nama_field});
contoh: alter table mahasiswa add primary key (NIM);
2

Menambah kolom
alter table {nama_table} add {nama_field_baru typedata({jumlah})} after {nama_field}; >bisa tanpa after<
contoh:alter table mahasiswa add agama varchar(12) after jurusan;
3

Merubah nama kolom
alter table {nama_table} change {kolom_lama} {kolom_baru} {typedata({jumlah})};
contoh: alter table mahasiswa change nama absen varchar(12);
4

Menghapus kolom
alter table {nama_table} drop {nama_kolom};
contoh: alter table mahasiswa drop nama;
5

Mengganti nama tabel
alter table {nama_tabel_lama} rename {nama_tabel_baru};
contoh: alter table mahasiswa rename mahastudent;



MEMASUKKAN DATA KEDALAM FIELD

1

Cara pertama
insert into {nama_tabel} ({field1}, {field2}, dst) values ('{data1}', '{data2}', dst);
contoh: insert into mahasiswa (NIM, nama, asal) values ('2021TI044', 'sukma', 'solo');
2

Cara kedua
insert into {nama_tabel} values ('{data1}', '{data2}', dst);
contoh: insert into mahasiswa values ('2021TI044', 'sukma', 'solo');
